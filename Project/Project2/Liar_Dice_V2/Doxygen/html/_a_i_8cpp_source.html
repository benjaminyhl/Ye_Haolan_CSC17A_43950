121 char AI::getMtFr() {
122  int *temp=new int[5];
123  int cnt=1;//count for the dice
124  int hgst;//highest number
125  int indx;//index
126  //if dices: 2 2 3 4 2, then temp: 3 3 1 1 3
127  for(int i=0;i<5;i++) {
128  for(int j=0;j<5;j++) {
129  if(dices[i]==dices[j]) cnt++;
130  }
131  temp[i]=cnt;
132  cnt=1;
133  }
134  hgst=temp[0];//initialize the highest number
135  indx=0;//initialize the index
136  //find out the highest and its index
137  for(int i=0;i<5;i++) {
138  if(temp[i]>hgst) {
139  hgst=temp[i];
140  indx=i;
141  }
142  }
143  delete []temp;//delete allocate memory
144  return dices[indx];
145 }
